 1import java.util.ArrayList;
 2import org.apache.commons.math3.optim.OptimizationData;
 3import org.apache.commons.math3.optim.PointValuePair;
 4import org.apache.commons.math3.optim.linear.LinearConstraint;
 5import org.apache.commons.math3.optim.linear.LinearConstraintSet;
 6import org.apache.commons.math3.optim.linear.LinearObjectiveFunction;
 7import org.apache.commons.math3.optim.linear.NonNegativeConstraint;
 8import org.apache.commons.math3.optim.linear.Relationship;
 9import org.apache.commons.math3.optim.linear.SimplexSolver;
10import org.apache.commons.math3.optim.nonlinear.scalar.GoalType;
11
12
13/*
14* @author  Jorge Garcia de Alba
15 */
16public class LPModel {
17
18    public static final int GEQ = 1;
19    public static final int LEQ = 2;
20    public static final int EQ = 3;
21
22    private LinearObjectiveFunction linearObjectiveFunction;
23    private ArrayList constraints;
24    private double[] point;
25    private double cost;
26    private LPFormat lpFormat;
27
28    public LPModel() {
29
30        //commons-math3-3.6.1 library on classpath
31        constraints = new ArrayList();
32        lpFormat = new LPFormat();
33    }
34
35    public void addObjective(double[] coefficients) {
36
37        byte constantTerm = 0;
38        //Objective function
39        linearObjectiveFunction = new LinearObjectiveFunction(coefficients, constantTerm);
40        //
41        lpFormat.objectiveToLp(coefficients);
42    }
43
44    public void addConstraint(double[] coefficients, int rel, double amount) {
45
46        Relationship relationship = null;
47
48        switch (rel) {
49            case LPModel.GEQ:
50                relationship = Relationship.GEQ;
51                //
52                break;
53            case LPModel.LEQ:
54                //
55                relationship = Relationship.LEQ;
56                break;
57            case LPModel.EQ:
58                //
59                relationship = Relationship.EQ;
60                break;
61            default:
62                relationship = Relationship.GEQ;
63                break;
64        }
65
66        //Linear Constraint
67        LinearConstraint c = new LinearConstraint(coefficients, relationship, amount);
68        constraints.add(c);
69        //
70        lpFormat.constraintToLp(coefficients,rel,amount);
71    }
72
73    public void solve() {
74
75        //Constraint Set
76        LinearConstraintSet linearConstraintSet = new LinearConstraintSet(constraints);
77
78        //Solution
79        GoalType minimize = GoalType.MINIMIZE;
80        NonNegativeConstraint nonNegativeConstraint = new NonNegativeConstraint(true);
81        PointValuePair solution = (new SimplexSolver()).optimize(new OptimizationData[]{linearObjectiveFunction, linearConstraintSet, minimize, nonNegativeConstraint});
82        point = solution.getPoint();
83        cost = solution.getSecond();
84    }
85
86    public double[] getPoint() {
87        return point;
88    }
89
90    public double getCost() {
91        return cost;
92    }
93
94    public String printModel(){
95        return lpFormat.getModel();
96    }
97}
